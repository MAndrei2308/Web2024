CREATE TABLE Users (
    user_id INT PRIMARY KEY AUTO_INCREMENT,
    username VARCHAR(50) UNIQUE,
    email VARCHAR(100) UNIQUE,
    password_hash VARCHAR(255) -- hash-ul parolei
);

CREATE TABLE Profiles (
    profile_id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT,
    first_name VARCHAR(50),
    last_name VARCHAR(50),
    email VARCHAR(100),
    -- Alte coloane pentru preferin»õe de suveniruri etc.
    FOREIGN KEY (user_id) REFERENCES Users(user_id)
);


CREATE TABLE Regions (
    region_id INT PRIMARY KEY AUTO_INCREMENT,
    region_name VARCHAR(100),
    description TEXT,
    image_url VARCHAR(255)
);

CREATE TABLE Visits (
    visit_id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT,
    region_id INT,
    visit_date DATE,
    FOREIGN KEY (user_id) REFERENCES Users(user_id),
    FOREIGN KEY (region_id) REFERENCES Regions(region_id)
);

CREATE TABLE Cultures (
    culture_id INT PRIMARY KEY AUTO_INCREMENT,
    culture_name VARCHAR(100),
    description TEXT,
    region_id INT,
    FOREIGN KEY (region_id) REFERENCES Regions(region_id)
);

CREATE TABLE Souvenirs (
    souvenir_id INT PRIMARY KEY AUTO_INCREMENT,
    souvenir_name VARCHAR(100),
    description TEXT,
    region_id INT,
    image_url VARCHAR(255),
    FOREIGN KEY (region_id) REFERENCES Regions(region_id)
);
